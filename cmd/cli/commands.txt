export DATABASE_URL=postgres://wreker:pwddd@localhost:5432/marketdb?sslmode=disable

# then freely use

./market add --name "Orange" --price 56.5 --calories 2395 --sugar 121
./market list
./market get --name "Apple"
./market delete --name "Apple"
./market list
./market update --name "Apple" --price 1.5 --calories 95 --sugar 145
./market update --name "Banana" --price 2.5 --calories 100 --sugar 110
./market list
